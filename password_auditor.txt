<script>
(function(){
  const COMMON_PASSWORDS = new Set(["password","123456","password1","12345678","qwerty","abc123","monkey","1234567","letmein","trustno1","dragon","baseball","iloveyou","master","sunshine","ashley","bailey","passw0rd","shadow","123123","654321","superman","qazwsx","michael","football","password123","admin","welcome","login","hello","charlie","donald","password2","qwerty123","1q2w3e4r","zaq1zaq1","123qwe","passw0rd1","pa55w0rd","pass@123","Passw0rd","Summer2023","Winter2023","Spring2023","Autumn2023","January2023","February2023","December2023"]);

  const KEYBOARD_ROWS = ["qwertyuiop","asdfghjkl","zxcvbnm","1234567890"];
  const SEQS = "abcdefghijklmnopqrstuvwxyz0123456789";

  const pw = document.getElementById("pa-pw");
  const fill = document.getElementById("pa-fill");
  const label = document.getElementById("pa-label");
  const scoreLabel = document.getElementById("pa-score-label");
  const stats = document.getElementById("pa-stats");
  const entropyEl = document.getElementById("pa-entropy");
  const lengthEl = document.getElementById("pa-length");
  const charsetEl = document.getElementById("pa-charset");
  const crackEl = document.getElementById("pa-crack");
  const patternsEl = document.getElementById("pa-patterns");
  const patSec = document.getElementById("pa-patterns-section");
  const breachSec = document.getElementById("pa-breach-section");
  const dot = document.getElementById("pa-dot");
  const breachMsg = document.getElementById("pa-breach-msg");
  const crackNote = document.getElementById("pa-crack-note");

  document.getElementById("pa-show").addEventListener("click", () => {
    pw.type = pw.type === "password" ? "text" : "password";
  });

  let debounceTimer;

  function charsetSize(p) {
    let s = 0;
    if (/[a-z]/.test(p)) s += 26;
    if (/[A-Z]/.test(p)) s += 26;
    if (/[0-9]/.test(p)) s += 10;
    if (/[^a-zA-Z0-9]/.test(p)) s += 32;
    return s;
  }

  function entropy(p) {
    const cs = charsetSize(p);
    if (cs === 0 || p.length === 0) return 0;
    return Math.round(p.length * Math.log2(cs));
  }

  function formatCrackTime(bits) {
    const guessesPerSecond = 1e10;
    const totalGuesses = Math.pow(2, bits);
    const seconds = totalGuesses / guessesPerSecond / 2;
    if (seconds < 1) return "< 1 second";
    if (seconds < 60) return Math.round(seconds) + " seconds";
    if (seconds < 3600) return Math.round(seconds/60) + " minutes";
    if (seconds < 86400) return Math.round(seconds/3600) + " hours";
    if (seconds < 31536000) return Math.round(seconds/86400) + " days";
    if (seconds < 3.156e9) return Math.round(seconds/31536000) + " years";
    if (seconds < 3.156e12) return (seconds/3.156e9).toFixed(1) + "k years";
    return "> 1 trillion years";
  }

  function detectPatterns(p) {
    const issues = [];
    const good = [];
    const pl = p.toLowerCase();

    if (p.length < 8) issues.push({t:"bad", l:"Too short (< 8 chars)"});
    else if (p.length < 12) issues.push({t:"warn", l:"Short — aim for 12+"});
    else good.push({t:"ok", l:"Good length"});

    if (/[A-Z]/.test(p)) good.push({t:"ok", l:"Uppercase letters"});
    else issues.push({t:"warn", l:"No uppercase letters"});

    if (/[0-9]/.test(p)) good.push({t:"ok", l:"Numbers present"});
    else issues.push({t:"warn", l:"No numbers"});

    if (/[^a-zA-Z0-9]/.test(p)) good.push({t:"ok", l:"Special characters"});
    else issues.push({t:"warn", l:"No special characters"});

    const uniqueRatio = new Set(p.split("")).size / p.length;
    if (uniqueRatio < 0.4) issues.push({t:"bad", l:"High character repetition"});

    if (/(.)\1{2,}/.test(p)) issues.push({t:"bad", l:"Repeated characters (aaa…)"});

    for (const row of KEYBOARD_ROWS) {
      for (let i = 0; i <= row.length - 4; i++) {
        const chunk = row.slice(i, i+4);
        if (pl.includes(chunk) || pl.includes(chunk.split("").reverse().join(""))) {
          issues.push({t:"bad", l:"Keyboard walk detected"}); break;
        }
      }
    }

    for (let i = 0; i <= SEQS.length - 4; i++) {
      const fwd = SEQS.slice(i, i+4);
      const rev = fwd.split("").reverse().join("");
      if (pl.includes(fwd) || pl.includes(rev)) {
        issues.push({t:"warn", l:"Sequential pattern (abc, 123…)"}); break;
      }
    }

    if (/^[a-zA-Z]+\d{1,4}[!@#$]?$/.test(p)) issues.push({t:"warn", l:"Word + numbers pattern"});
    if (/(\d)\1{3,}/.test(p)) issues.push({t:"bad", l:"Repeated digits"});
    if (/^(.+)\1+$/.test(p)) issues.push({t:"bad", l:"Repeated substring"});

    const hasleet = /[4@3310!5$]/.test(p);
    if (hasleet && /[aAeEiIoOsSbBtTlL]/.test(p)) issues.push({t:"warn", l:"Leet substitution (predictable)"});

    return [...issues, ...good];
  }

  async function sha1(str) {
    const buf = new TextEncoder().encode(str);
    const hash = await crypto.subtle.digest("SHA-1", buf);
    return Array.from(new Uint8Array(hash)).map(b => b.toString(16).padStart(2,"0")).join("").toUpperCase();
  }

  let lastPw = "";

  async function checkBreach(p) {
    dot.className = "pa-dot checking";
    breachMsg.textContent = "Checking breach databases…";

    if (COMMON_PASSWORDS.has(p.toLowerCase())) {
      dot.className = "pa-dot common";
      breachMsg.textContent = "Found in common password list — extremely risky";
      return;
    }

    try {
      const hash = await sha1(p);
      const prefix = hash.slice(0,5);
      const suffix = hash.slice(5);
      const res = await fetch(`https://api.pwnedpasswords.com/range/${prefix}`, {
        headers: {"Add-Padding": "true"}
      });
      if (!res.ok) throw new Error("API error");
      const text = await res.text();
      const line = text.split("\n").find(l => l.startsWith(suffix));
      if (line) {
        const count = parseInt(line.split(":")[1]);
        dot.className = "pa-dot pwned";
        breachMsg.textContent = `Found in ${count.toLocaleString()} data breaches — do not use`;
      } else {
        dot.className = "pa-dot safe";
        breachMsg.textContent = "Not found in known breach databases";
      }
    } catch(e) {
      dot.className = "pa-dot checking";
      breachMsg.textContent = "Breach check unavailable (network restricted)";
    }
  }

  function scoreColor(bits) {
    if (bits < 28) return "#E24B4A";
    if (bits < 45) return "#EF9F27";
    if (bits < 65) return "#FAC775";
    if (bits < 90) return "#639922";
    return "#1D9E75";
  }

  function scoreText(bits, p) {
    if (COMMON_PASSWORDS.has(p.toLowerCase())) return ["Very weak", 5];
    if (bits < 28) return ["Very weak", 10];
    if (bits < 45) return ["Weak", 30];
    if (bits < 65) return ["Fair", 55];
    if (bits < 90) return ["Strong", 78];
    return ["Very strong", 100];
  }

  function charsetLabel(p) {
    const parts = [];
    if (/[a-z]/.test(p)) parts.push("a-z");
    if (/[A-Z]/.test(p)) parts.push("A-Z");
    if (/[0-9]/.test(p)) parts.push("0-9");
    if (/[^a-zA-Z0-9]/.test(p)) parts.push("sym");
    return parts.join("+") || "—";
  }

  pw.addEventListener("input", () => {
    const p = pw.value;
    if (!p) {
      fill.style.width = "0%"; fill.style.background = "#888";
      label.textContent = "—"; scoreLabel.textContent = "";
      stats.style.display = "none";
      patSec.style.display = "none";
      breachSec.style.display = "none";
      return;
    }

    const bits = entropy(p);
    const [sText, pct] = scoreText(bits, p);
    const col = COMMON_PASSWORDS.has(p.toLowerCase()) ? "#E24B4A" : scoreColor(bits);

    fill.style.width = pct + "%";
    fill.style.background = col;
    label.textContent = sText;
    scoreLabel.style.color = col;
    scoreLabel.textContent = bits + " bits";

    stats.style.display = "grid";
    entropyEl.textContent = bits + " bits";
    lengthEl.textContent = p.length + " chars";
    charsetEl.textContent = charsetLabel(p);
    crackEl.textContent = formatCrackTime(bits);

    const tags = detectPatterns(p);
    patternsEl.innerHTML = tags.map(t =>
      `<span class="pa-tag ${t.t}">${t.t==="ok"?"✓":t.t==="warn"?"⚠":"✗"} ${t.l}</span>`
    ).join("");
    patSec.style.display = "block";

    breachSec.style.display = "block";
    crackNote.textContent = `Estimated crack time assumes 10 billion guesses/sec (GPU cluster).`;

    clearTimeout(debounceTimer);
    debounceTimer = setTimeout(() => {
      if (pw.value === p) checkBreach(p);
    }, 600);
  });
})();
</script>
